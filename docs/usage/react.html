# 在 React 中使用

下面的 React SFC 左侧可以编辑, 右侧能实时渲染 (如果遇到加载过慢的情况下, 可以考虑搭梯子或耐心等待)


# 方式 1：通过 import 引入

  1. 首先安装插件
# npm 安装：
npm install @lucky-canvas/react@latest

# yarn 安装：
yarn add @lucky-canvas/react@latest
  1. 然后在你的组件中引入并使用
import { LuckyWheel, LuckyGrid } from '@lucky-canvas/react'

export default function Test () {
  return <div>

    // 大转盘抽奖
    <LuckyWheel width="300px" height="300px" ...你的配置></LuckyWheel>

    // 九宫格抽奖
    <LuckyGrid width="300px" height="300px" ...你的配置></LuckyGrid>

    // 老虎机抽奖
    <SlotMachine width="300px" height="300px" ...你的配置></SlotMachine>

  </div>
}
