# 在 Js / JQ 中使用

# 方式 1：通过 script 标签引入

  • 指定版本：https://cdn.jsdelivr.net/npm/lucky-canvas@1.7
<div id="my-lucky"></div>

<script src="https://cdn.jsdelivr.net/npm/lucky-canvas@1.7"></script>
<script>
  const myLucky = new LuckyCanvas.LuckyWheel('#my-lucky', {
    width: '200px',
    height: '200px',
    blocks: [{ padding: '10px', background: '#869cfa' }],
    prizes: [
      { fonts: [{ text: '0' }], background: '#e9e8fe' },
      { fonts: [{ text: '1' }], background: '#b8c5f2' },
      { fonts: [{ text: '2' }], background: '#e9e8fe' },
      { fonts: [{ text: '3' }], background: '#b8c5f2' },
      { fonts: [{ text: '4' }], background: '#e9e8fe' },
      { fonts: [{ text: '5' }], background: '#b8c5f2' },
    ],
  })
</script>
